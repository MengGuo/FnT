\section*{Workspace, logic, and automata}
\begin{center}
\begin{tabularx}{\textwidth}{>{\bfseries}p{0.24\textwidth} X}
\toprule
Symbol & Meaning \\
\midrule
$\mathcal{W}$ & Workspace. \\
$\Pi=\{\pi_1,\ldots,\pi_N\}$ & Finite partition of the workspace; each $\pi_i$ is a discrete region. \\
$AP$ & Set of atomic propositions. \\
$L_c$ & Labeling function on the discrete transition system. \\
$\mathcal{T}_c$ & Control-driven weighted finite-state transition system. \\
$\longrightarrow_c$ & Transition relation of $\mathcal{T}_c$. \\
$w_c$ & Transition-cost function on $\longrightarrow_c$. \\
$\tau$ & Discrete robot path or motion plan on the transition system. \\
$\texttt{trace}(\tau)$ & Word generated by the path $\tau$. \\
$\varphi$ & Temporal-logic task specification; sc-LTL in later chapters unless stated otherwise. \\
$\mathcal{A}_{\varphi}$ & Automaton associated with $\varphi$. \\
$\mathcal{A}_p$ & Product automaton between $\mathcal{T}_c$ and $\mathcal{A}_{\varphi}$. \\
$R$ & Run on a product automaton. \\
\bottomrule
\end{tabularx}
\end{center}

\section*{Local tasks, teams, and plans}
\begin{center}
\begin{tabularx}{\textwidth}{>{\bfseries}p{0.24\textwidth} X}
\toprule
Symbol & Meaning \\
\midrule
$\mathcal{N}=\{1,\ldots,N\}$ & Robot index set. \\
$\mathcal{A}_i$ & Primitive-action set of robot $i$. \\
$\Phi=\{\varphi_1,\ldots,\varphi_M\}$ & Mission set. \\
$\omega_{m,k}$ & $k$th collaborative task associated with mission $\varphi_m$. \\
$\eta_{m,k}(\cdot)$ & Required execution duration of collaborative task component $\omega_{m,k}$. \\
$\Xi_{m,k}$ & Required subtask set associated with $\omega_{m,k}$. \\
$\tau_i$ & Nominal timed plan of robot $i$. \\
$\Pi_{\mathrm{team}}$ & Team-level finite motion-and-action plan in the top-down synthesis framework. \\
$\Gamma_k$ & Task sequence assigned to team $k$. \\
$\tau_i^{\mathrm{tsk}}$ & Robot-level task sequence induced from the team task sequence $\Gamma_k$. \\
$\widetilde{\tau}_i^{\ell}$ & Local execution plan of robot $i$ for the $\ell$th collaborative task handled by its team. \\
$\nu$ & Search-tree node or partial assignment object in the corresponding chapter-specific planning routine. \\
$\eta_{\nu}$ & Concurrency index used in the team-synthesis search of Chapter~4. \\
\bottomrule
\end{tabularx}
\end{center}

\section*{Communication-aware coordination and dynamic tasking}

\midrule

$\eta(J)$ & Long-run average task-completion rate under the collective plan $J$. \\
$\Gamma_i$ & Communication-aware local plan of robot $i$. \\
$\widehat{\Gamma}$ & Joint communication-aware plan of the full operator--fleet system. \\
$\xi^{\ell}$ & Incoming request of type $\ell$ in the online request-handling framework. \\

$\widehat{\pi}_k^{\xi}$ & Local path associated with the graph instantiation for request $\xi$. \\
$\Pi_k^{\xi}$ & Path family or path set associated with request $\xi$ in the graph-instantiation step. \\

\section*{Uncertainty, prediction, and continuous trajectories}
\begin{center}
\begin{tabularx}{\textwidth}{>{\bfseries}p{0.24\textwidth} X}
\toprule
Symbol & Meaning \\
\midrule
$x_t$ & Joint robot state at discrete time $t$ in Chapter~6. \\
$Y_t$ & Joint target-state vector at time $t$. \\
$V_t$ & Joint target-velocity vector at time $t$. \\
$D_{\texttt{cal}},D_{\texttt{tra}}$ & Calibration and training trajectory datasets for target-motion prediction. \\
$\boldsymbol{\tau}_{\mathcal{N}}$ & Team task plan in the uncertain open-world setting. \\
$\tau_n$ & Timed action-target sequence of robot $n$ in the team task plan. \\
$w_{\tau}$ & Finite trace induced by the team task plan $\boldsymbol{\tau}_{\mathcal{N}}$. \\
$P_{\varphi}=(\Omega_{\varphi},\preceq_{\varphi},\neq_{\varphi})$ & R-poset associated with the task formula $\varphi$. \\
$\Omega_{\varphi}$ & Subtask set induced from the task formula. \\
$\preceq_{\varphi}$ & Precedence relation in the R-poset. \\
$\neq_{\varphi}$ & Incompatibility relation in the R-poset. \\
$\overline{T}_{\varphi}$ & Average makespan metric for the released tasks. \\
$\mathbf{p}_i(t)$ & Position of robot $i$ at time $t$. \\
$\mathbf{v}_i(t)$ & Velocity of robot $i$ at time $t$. \\
$\mathbf{q}_i(t)$ & State of robot $i$, typically $(\mathbf{p}_i(t),\mathbf{v}_i(t))$. \\
$\mathbf{Z}_i(T)$ & Finite-horizon trajectory of robot $i$. \\
$\Gamma_{\ell}$ & Partial collective trajectory set used in the sequential trajectory-construction procedure. \\
\bottomrule
\end{tabularx}
\end{center}
